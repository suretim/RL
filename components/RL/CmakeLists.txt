set(TFLITE_META_SRCS "esp_32_tflite_ewc_hvac_agent.cc" "../../fed_server/models/meta_model.tflite")

set(APPLICATION_SRCS
    "hvac_q_agent.cc"

    "config_wifi.cc"
    "config_spiffs.cc"
    "classifier_storage.cc"

    "model.pb.c"

# 主组件注册

    SRCS 
        "${APPLICATION_SRCS}"

    INCLUDE_DIRS 
        "."

    PRIV_INCLUDE_DIRS 
        "."

    EMBED_FILES 
        "${TFLITE_META_SRCS}"

        esp_wifi
        esp_netif

        json
        esp_timer
            
)

# C++ 特定设置
target_compile_options(${COMPONENT_LIB} PRIVATE 
    "-Wno-unused-variable"
    "-Wno-unused-but-set-variable"
    "-std=c++11"
)

    CONFIG_SPIRAM_USE=y
    CONFIG_SPIRAM_IGNORE_NOTFOUND=y
